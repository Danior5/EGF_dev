#Staliongrad
	on_capitulation = { # STG Kills STL
		effect = {
			if = {
				limit = {
					ROOT = {
						tag = STL
					}
					FROM = {
						tag = STG
					}
				}
				STG = {
					save_global_event_target_as = winning_country
				}
				STL = {
					save_global_event_target_as = losing_country
				}
				STG = {
					annex_country = {
						target = STL
						transfer_troops = no
					}
				}
				STG = {
					add_named_threat = {
						threat = 1
						name = "Scripted peace for [STL.GetName]"
					}
				}
				#event = scriptpeace.1
				STG = {
					clear_global_event_target = winning_country
				}
				STL = {
					clear_global_event_target = losing_country
				}
			}
		}
	}
