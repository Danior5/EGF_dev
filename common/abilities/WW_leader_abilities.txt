ability = {
	WNG_probing_attack = {
		name = WNG_ABILITY_PROBING_ATTACK
		desc = WNG_ABILITY_PROBING_ATTACK_DESC
		
		type = army_leader
		
		allowed = {
            OWNER = {
                has_country_flag = WW_WNG_special_ability_enabled
            }
			is_border_war = no
		}
		
		unit_modifiers = {
			offence = -0.2
		}
		if = {
			limit = {
				OWNER = {
					NOT = {
						has_country_flag = probing_attack_yes
					}
				}
			}
			OWNER = {
				set_country_flag = probing_attack_yes
			}
			add_temporary_buff_to_units = {
				org_damage_multiplier = 0.25
				str_damage_multiplier = -0.4
				days = -1
			}
		}
		else = {
			add_temporary_buff_to_units = {
				org_damage_multiplier = -0.25
				str_damage_multiplier = 0.4
				days = -1
			}
			OWNER = {
				clear_country_flag = probing_attack_yes
			}
		}

		cancelable = yes
		cost = 0.1
		duration = -1
	}

}
